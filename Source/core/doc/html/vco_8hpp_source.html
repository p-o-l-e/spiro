86 {
87 std::complex<float> fI = exp( std::complex<float> (0.0, 1.0) * carrier);
88 std::complex<float> fO = exp( std::complex<float> (0.0, 1.0) * modulator);
89 float fE = std::arg( fI * std::conj(fO) );
