67// Envelope Follower /////////////////////////////////////////////////////////////////////////////////////////////
68struct envelope_follower
69{
70 float a;
71 float r;
72 float envelope = 0.0f;
73
74 inline void init(const float&, const float&, const float&);
75 inline void process(const float& in);
76};
77
78inline void envelope_follower::init(const float& aMs, const float& rMs, const float& sample_rate )
79{
80 a = pow( 0.01, 1.0 / ( aMs * sample_rate * 0.001 ) );
81 r = pow( 0.01, 1.0 / ( rMs * sample_rate * 0.001 ) );
82}
83
84inline void envelope_follower::process(const float& in)
85{
86 float f = fabs(in);
87 if (f > envelope) envelope = a * ( envelope - f ) + f;
88 else envelope = r * ( envelope - f ) + f;
89}

112inline float limiter_t::process(const float& in)
113{
114 float out = in;
115 e.process(out);

118 out /= expf(e.envelope - threshold);

131// 3to2 //////////////////////////////////////////////////////////////////////////////////////////////////////////
132template <typename T>
133constexpr point2d<T> c3c2(const point3d<T>& in, const float& a, const float& b) noexcept
134{
135 auto ac = a > 0.5f ? (1.0f - a) * 2.0f : a * 2.0f;
136 auto bc = b > 0.5f ? (1.0f - b) * 2.0f : b * 2.0f;
137 return point2d<T> { xfade(in.x, in.y, ac), xfade(in.y, in.z, bc) };
138}
