135// Calculates the Hamilton product of two quaternions
136constexpr quaternion &quaternion::operator*=(const quaternion &q)
137{
138 float w1 = w;
139 float x1 = x;
140 float y1 = y;
141 float z1 = z;
142
143 float w2 = q.w;
144 float x2 = q.x;
145 float y2 = q.y;
146 float z2 = q.z;
147
148 w = w1 * w2 - x1 * x2 - y1 * y2 - z1 * z2;
149 x = w1 * x2 + x1 * w2 + y1 * z2 - z1 * y2;
150 y = w1 * y2 + y1 * w2 + z1 * x2 - x1 * z2;
151 z = w1 * z2 + z1 * w2 + x1 * y2 - y1 * x2;
152 return *this;
153}

190// Rotates a vector according to the current quaternion, assumes |q|=1
191// @link https://raw.org/proof/vector-rotation-using-quaternions/
192constexpr void quaternion::rotate_vector(float &vx, float &vy, float &vz)
193{
194 // t = 2q x v
195 float tx = 2. * (y * vz - z * vy);
196 float ty = 2. * (z * vx - x * vz);
197 float tz = 2. * (x * vy - y * vx);
198
199 // v + w t + q x t
200 vx = vx + w * tx + y * tz - z * ty;
201 vy = vy + w * ty + z * tx - x * tz;
202 vz = vz + w * tz + x * ty - y * tx;
203}

205// Creates a quaternion by a rotation given by Euler angles (multiplication order from right to left)
206constexpr void quaternion::from_euler(const float& _x, const float& _y, const float& _z)
207{
208 float X = _x * 0.5f;
209 float Y = _y * 0.5f;
210 float Z = _z * 0.5f;
211
212 float cX = cosf(X);
213 float cY = cosf(Y);
214 float cZ = cosf(Z);
215
216 float sX = sinf(X);
217 float sY = sinf(Y);
218 float sZ = sinf(Z);
219
220 // roll around X, pitch around Y, yaw around Z
221 // axisAngle([1, 0, 0], φ) * axisAngle([0, 1, 0], θ) * axisAngle([0, 0, 1], ψ)
222 w = cX * cY * cZ - sX * sY * sZ;
223 x = sX * cY * cZ + sY * sZ * cX;
224 y = sY * cX * cZ - sX * sZ * cY;
225 z = sX * sY * cZ + sZ * cX * cY;
226}

228// Creates quaternion by a rotation given as axis-angle orientation
229constexpr void quaternion::from_axis_angle(const float& _x, const float& _y, const float& _z, const float& angle)
230{
231 float halfAngle = angle * 0.5;
232
233 float sin_2 = sinf(halfAngle);
234 float cos_2 = cosf(halfAngle);
235
236 float sin_norm = sin_2 / sqrtf(_x * _x + _y * _y + _z * _z);
237
238 w = cos_2;
239 x = x * sin_norm;
240 y = y * sin_norm;
241 z = z * sin_norm;
242}

246template <typename T>
247constexpr point3d<T> RotateX(const point3d<T>& o, const T& theta) noexcept
248{
249 return point3d<T> { o.l, o.c * cosf(theta) - o.r * sinf(theta), o.c * sinf(theta) + o.r * cosf(theta)};
250}
251
252template <typename T>
253constexpr point3d<T> RotateY(const point3d<T>& o, const T& theta) noexcept
254{
255 return point3d<T> { o.l * cosf(theta) + o.r * sinf(theta), o.c, o.r * cosf(theta) - o.l * sinf(theta)};
256}
257
258template <typename T>
259constexpr point3d<T> RotateZ(const point3d<T>& o, const T& theta) noexcept
260{
261 return point3d<T> { o.l * cosf(theta) - o.c * sinf(theta), o.l * sinf(theta) + o.c * cosf(theta), o.r};
262}
263
264template <typename T>
265constexpr point3d<T> Rotate(const point3d<T>& o, const T& x, const T& y, const T& z) noexcept
266{
267 return RotateZ(RotateY(RotateX(o, x) , y), z);
268}
