130constexpr float capsule_sdf(float px, float py, float ax, float ay, float bx, float by, float r)
131{
132 float pax = px - ax, pay = py - ay, bax = bx - ax, bay = by - ay;
133 float h = fmaxf(fminf((pax * bax + pay * bay) / (bax * bax + bay * bay), 1.0f), 0.0f);
134 float dx = pax - bax * h, dy = pay - bay * h;
135 return sqrtf(dx * dx + dy * dy) - r;
136}

143constexpr void lineSDFAABB(core::frame<float>* canvas, float ax, float ay, float bx, float by, float radius, float alpha)
144{
145 int xo = (int)floorf(fminf(ax, bx) - radius);
146 int xe = (int) ceilf(fmaxf(ax, bx) + radius);
147 int yo = (int)floorf(fminf(ay, by) - radius);
148 int ye = (int) ceilf(fmaxf(ay, by) + radius);
149 for (int y = yo; y <= ye; y++)
150 for (int x = xo; x <= xe; x++)
151 alphablend(canvas, x, y, fmaxf(fminf(0.5f - capsule_sdf(x, y, ax, ay, bx, by, radius), 1.0f), alpha));
152}
